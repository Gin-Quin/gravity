## api

```tsx
//  /components/Add.tsx
import { api } from "/api"
import { createSignal } from "solid-js";

export function Add() {
  const [result, setResult] = createSignal<undefined | number>(undefined)

  api.math.add(1, 2).then(setResult)

  return (
    <>
      {result === undefined ? `Loading...` : result}
    </>
  )
}
```
